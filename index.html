 packages = ["requests", "bs4", "js"] import req import requests from bs4 import BeautifulSoup from js import document def getSource(): req.getSource() url = document.querySelector("input.source-url") output = document.querySelector("div.output-source") htmlPage = requests.get(url).content soup = BeautifulSoup(htmlPage, 'html.parser') htmlPage = soup.prettify() output.innerHTML = htmlPage
